#### 第0007题
有个目录，里面是你自己写过的程序，统计一下你写过多少行代码。包括空行和注释，但是要分别列出来。

```
import glob

path = glob.glob('d://exercises/banksys/*.py')
for p in path:
    with open(p, 'r', encoding="utf-8") as fr:
        frs = fr.readlines()
        count_space = 0
        count_note = 0
        for f in frs:
            if f.strip().startswith("#"):
                count_note += 1
            elif f.isspace():
                count_space += 1
    filename = p[22:]
    print('文件%s共有%s行代码，其中%s行是空白行，其中%s行是注释' %(filename, len(frs), count_space,count_note))
```

#### 第0008题
 一个HTML文件，找出里面的正文。

```
import re
import urllib.request

#爬取静态网页的源码
def spider():
    url = "https://pypi.org/project/lxml/"
    headers = {"User-Agent" : "Mozilla/5.0 (Windows NT 6.1; WOW64)"}
    request = urllib.request.Request(url, headers=headers)
    response = urllib.request.urlopen(request)
    content = response.read()
    return content.decode('utf-8')
html = spider()

#解析文件
def parse():
    pattern = re.compile(r'<div\sclass="project-description">(.*?)</div>', re.S)
    html_list = pattern.findall(html)
    for hl in html_list:
        write(hl.strip())
#正文写入txt文件
def write(item):
    with open('d://code/zhengwen.txt', 'a') as f:
        f.write(item)

parse()
```
