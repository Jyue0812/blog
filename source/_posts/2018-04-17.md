#### 第0005题
你有一个目录，装了很多照片，把它们的尺寸变成都不大于 iPhone5 分辨率的大小。

```
#导入glob和pillow库
import glob
from PIL import Image

#设置iPhone5的分辨率高度宽度
iphone5_width = 1136
iphone5_depth = 640

path = glob.glob('*.jpg')
for p in path:
    image = Image.open(p, 'r')
    #取图片高宽
    width, height = image.size
	
	#对图片宽度高度与iPhone5高度宽度进行比较
    if width > iphone5_width:
        new_width = iphone5_width * height / width
        new_height = iphone5_depth * height / width
	
	if height > iphone5_height:
        new_width = iphone5_width * height / width
        new_height = iphone5_depth * height / width
        
        #重新等比例缩小图片分辨率
        new_image = image.resize((int(new_width), int(new_height)))
        #以新的名字命名并保存
        new_image_name = p[:-4] + '_new.jpg'
        new_image.save(new_image_name, 'jpeg')
```
